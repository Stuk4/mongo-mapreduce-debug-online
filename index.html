{
    map: function() {
        emit({
                customer: this.cust_id,
                month: this.ord_date.getUTCMonth()
            },
            this.price);
    },

    reduce: function(key, values) {
        return Array.sum(values);
    },

    finalize: function(key, reduced) {
        return reduced;
    }
}
